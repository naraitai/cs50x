
Prompt user for input.
Reprompt until gets string

store input string

create letter counter
loop input string
    if item is alphabetical character (ctype.h > int isalpha(); 0 means non-alphabetical. Above 0 means alphabetical.)
        add 1 to letter counter
    else
        do nothing

create word counter
loop input string
    if space is found '\0' (ctype.h > int isspace(); 0 means non-whitespace. Above 0 means whitespace.)
        add 1 to word counter
    else
        do nothing
add 1 to word counter

create sentence counter
loop input string
    if fullstop
        add 1 to sentence counter
    else if exclamation
        add 1 to sentence counter
    else if question mark
        add 1 to sentence counter
    else
        do nothing

index calculation (0.0588 * letter count - 0.296 * sentence count - 15.8)

round index calculation to nearest integer value

if round index is less than 1
    display "Before Grade 1"
else if round index is 16 or more
    display "Grade 16+"
else
    display "Grade" + round index
